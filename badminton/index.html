Can we simulate the trajectory ?¶

Newton’s second law:¶

def derivative(X, t):
    """
    Target ODE: Newton's second law
    """
    x, y, vx, vy = X
    v = (vx**2 + vy**2) ** 0.5
    Tx, Ty = vx / v, vy / v
    ax = -0.5 * rho * v**2 * A * cx * Tx / m
    ay = -0.5 * rho * v**2 * A * cx * Ty / m - g
    return np.array([vx, vy, ax, ay])

theta0 = 45.0

t = np.linspace(0.0, 10.0, 200)
sol = integrate.odeint(derivative, X0, t)
